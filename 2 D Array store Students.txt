4. Store and display marks of students in multiple subjects using a 2D array.


#include <iostream>
using namespace std;

int main() {
    int students, subjects;
    int marks[10][10];

    cout << "Enter number of students: ";
    cin >> students;

    cout << "Enter number of subjects: ";
    cin >> subjects;

    // Input marks
    for (int i = 0; i < students; i++) {
        cout << "\nEnter marks for Student " << i + 1 << ":\n";
        for (int j = 0; j < subjects; j++) {
            cout << "Subject " << j + 1 << ": ";
            cin >> marks[i][j];
        }
    }

    // Display marks
    cout << "\n--- Student Marks ---\n";
    for (int i = 0; i < students; i++) {
        cout << "Student " << i + 1 << ": ";
        for (int j = 0; j < subjects; j++) {
            cout << marks[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
