1. Create an array to store student names in a class, allowing insertion, deletion, and traversal
operations.


#include <iostream>
#include <string>
using namespace std;

int main() {
    string students[50];   // array to store names
    int n = 0;             // current number of students
    int choice;

    do {
        cout << "\n--- Student Name Array Menu ---\n";
        cout << "1. Insert Student Name\n";
        cout << "2. Delete Student Name\n";
        cout << "3. Display Student Names\n";
        cout << "4. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        if (choice == 1) {
            cout << "Enter student name: ";
            cin >> students[n];
            n++;
        }
        else if (choice == 2) {
            int pos;
            cout << "Enter position to delete (0 to " << n - 1 << "): ";
            cin >> pos;

            if (pos < 0 || pos >= n) {
                cout << "Invalid position!";
            } else {
                for (int i = pos; i < n - 1; i++) {
                    students[i] = students[i + 1];
                }
                n--;
                cout << "Student deleted successfully.";
            }
        }
        else if (choice == 3) {
            cout << "Student Names:\n";
            for (int i = 0; i < n; i++) {
                cout << i << ". " << students[i] << endl;
            }
        }

    } while (choice != 4);

    return 0;
}
