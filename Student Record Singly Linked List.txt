#include <iostream>
using namespace std;

struct Student {
    int roll;
    string name;
    Student* next;
};

Student* head = NULL;

// Create / Insert student record
void insertStudent() {
    Student* newNode = new Student;
    cout << "Enter Roll No: ";
    cin >> newNode->roll;
    cout << "Enter Name: ";
    cin >> newNode->name;

    newNode->next = head;
    head = newNode;
}

// Delete student record by roll number
void deleteStudent() {
    int roll;
    cout << "Enter Roll No to delete: ";
    cin >> roll;

    Student* temp = head;
    Student* prev = NULL;

    while (temp != NULL && temp->roll != roll) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL) {
        cout << "Student not found!\n";
        return;
    }

    if (prev == NULL) {
        head = temp->next;
    } else {
        prev->next = temp->next;
    }

    delete temp;
    cout << "Student record deleted.\n";
}
1. Implement a student record system using a singly linked list that supports creating, inserting, deleting, and displaying student records.


// Display all student records
void displayStudents() {
    Student* temp = head;
    cout << "\nStudent Records:\n";
    while (temp != NULL) {
        cout << "Roll: " << temp->roll << ", Name: " << temp->name << endl;
        temp = temp->next;
    }
}

int main() {
    int choice;

    do {
        cout << "\n1.Insert  2.Delete  3.Display  4.Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        if (choice == 1)
            insertStudent();
        else if (choice == 2)
            deleteStudent();
        else if (choice == 3)
            displayStudents();

    } while (choice != 4);

    return 0;
}
