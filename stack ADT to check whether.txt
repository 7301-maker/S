2. Use the stack ADT to check whether parentheses in an arithmetic expression are balanced.


#include <iostream>
#include <stack>
using namespace std;

// Function to check balanced parentheses
bool isBalanced(string expr) {
    stack<char> s;

    for (int i = 0; i < expr.length(); i++) {
        char ch = expr[i];

        // Push opening brackets
        if (ch == '(' || ch == '{' || ch == '[') {
            s.push(ch);
        }
        // Check closing brackets
        else if (ch == ')' || ch == '}' || ch == ']') {
            if (s.empty())
                return false;

            char top = s.top();
            s.pop();

            if ((ch == ')' && top != '(') ||
                (ch == '}' && top != '{') ||
                (ch == ']' && top != '['))
                return false;
        }
    }

    return s.empty();
}

int main() {
    string expr;

    cout << "Enter arithmetic expression: ";
    cin >> expr;

    if (isBalanced(expr))
        cout << "Parentheses are BALANCED\n";
    else
        cout << "Parentheses are NOT BALANCED\n";

    return 0;
}
