2. A university stores student roll numbers in a sorted list. Implement binary search to find whether a given roll number exists in the list.


#include <iostream>
using namespace std;

int main() {
    int n;
    int roll[50];
    int key;
    int low, high, mid;
    bool found = false;

    cout << "Enter number of students: ";
    cin >> n;

    cout << "Enter roll numbers (sorted):\n";
    for (int i = 0; i < n; i++) {
        cin >> roll[i];
    }

    cout << "Enter roll number to search: ";
    cin >> key;

    low = 0;
    high = n - 1;

    while (low <= high) {
        mid = (low + high) / 2;

        if (roll[mid] == key) {
            cout << "Roll number found at position " << mid;
            found = true;
            break;
        }
        else if (roll[mid] < key) {
            low = mid + 1;
        }
        else {
            high = mid - 1;
        }
    }

    if (!found) {
        cout << "Roll number NOT found";
    }

    return 0;
}
