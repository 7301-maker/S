5. Implement the Bubble Sort algorithm to sort an array of student marks.


#include <iostream>
using namespace std;

int main() {
    int n;
    int marks[50];

    cout << "Enter number of students: ";
    cin >> n;

    cout << "Enter student marks:\n";
    for (int i = 0; i < n; i++) {
        cin >> marks[i];
    }

    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (marks[j] > marks[j + 1]) {
                int temp = marks[j];
                marks[j] = marks[j + 1];
                marks[j + 1] = temp;
            }
        }
    }

    cout << "Sorted student marks:\n";
    for (int i = 0; i < n; i++) {
        cout << marks[i] << " ";
    }

    return 0;
}
