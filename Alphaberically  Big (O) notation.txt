3. Implement a program to analyze the performance of sorting student names alphabetically and measure its time complexity using Big O notation.

#include <iostream>
using namespace std;

int main() {
    int n;
    string students[50];

    cout << "Enter number of students: ";
    cin >> n;

    cout << "Enter student names:\n";
    for (int i = 0; i < n; i++) {
        cin >> students[i];
    }

    // Bubble Sort for alphabetical order
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (students[j] > students[j + 1]) {
                string temp = students[j];
                students[j] = students[j + 1];
                students[j + 1] = temp;
            }
        }
    }

    cout << "\nSorted Student Names:\n";
    for (int i = 0; i < n; i++) {
        cout << students[i] << endl;
    }

    return 0;
}
