3. Merge two linked lists containing student names from two different classes into a single list.

#include <iostream>
using namespace std;

// Node structure
struct Node {
    string name;
    Node* next;
};

// Insert node at end
void insertEnd(Node* &head, string name) {
    Node* newNode = new Node();
    newNode->name = name;
    newNode->next = NULL;

    if (head == NULL) {
        head = newNode;
        return;
    }

    Node* temp = head;
    while (temp->next != NULL) {
        temp = temp->next;
    }
    temp->next = newNode;
}

// Merge two linked lists
Node* mergeLists(Node* head1, Node* head2) {
    if (head1 == NULL)
        return head2;

    Node* temp = head1;
    while (temp->next != NULL) {
        temp = temp->next;
    }
    temp->next = head2;

    return head1;
}

// Display linked list
void display(Node* head) {
    Node* temp = head;
    while (temp != NULL) {
        cout << temp->name << " -> ";
        temp = temp->next;
    }
    cout << "NULL\n";
}

int main() {
    Node* classA = NULL;
    Node* classB = NULL;

    int n, m;
    string name;

    cout << "Enter number of students in Class A: ";
    cin >> n;
    cout << "Enter names of Class A students:\n";
    for (int i = 0; i < n; i++) {
        cin >> name;
        insertEnd(classA, name);
    }

    cout << "\nEnter number of students in Class B: ";
    cin >> m;
    cout << "Enter names of Class B students:\n";
    for (int i = 0; i < m; i++) {
        cin >> name;
        insertEnd(classB, name);
    }

    cout << "\nClass A List:\n";
    display(classA);

    cout << "Class B List:\n";
    display(classB);

    Node* mergedList = mergeLists(classA, classB);

    cout << "\nMerged Student List:\n";
    display(mergedList);

    return 0;
}
